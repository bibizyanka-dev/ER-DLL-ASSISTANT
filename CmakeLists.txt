cmake_minimum_required(VERSION 3.20)
project(EldenRingImGuiTrainer LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

file(GLOB IMGUI_SOURCES
    "ImGui/*.cpp"
    "ImGui/*.h"
)

file(GLOB MINHOOK_SOURCES
    "MinHook/src/*.c"
    "MinHook/src/*.h"
    "MinHook/src/HDE/*.c"
    "MinHook/src/HDE/*.h"
)

file(GLOB PROJECT_SOURCES
    "main.cpp"
    "src/d3d12/*.cpp"
    "src/imgui/*.cpp"
    "src/trainer/*.cpp"
    "src/utils/*.cpp"
)

add_library(MiniEldenTrainer SHARED
    ${PROJECT_SOURCES}
    ${IMGUI_SOURCES}
    ${MINHOOK_SOURCES}
)

target_include_directories(MiniEldenTrainer PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/ImGui
    ${CMAKE_CURRENT_SOURCE_DIR}/MinHook/include
    ${CMAKE_CURRENT_SOURCE_DIR}/MinHook/src
    ${CMAKE_CURRENT_SOURCE_DIR}/MinHook/src/HDE
)

target_link_libraries(MiniEldenTrainer PRIVATE
    d3d12
    dxgi
    d3dcompiler
)

target_compile_definitions(MiniEldenTrainer PRIVATE
    WIN32_LEAN_AND_MEAN
    NOMINMAX
)

target_compile_options(MiniEldenTrainer PRIVATE
    $<$<CONFIG:Release>:/O2 /MT /GS- /W3>
    $<$<CONFIG:Debug>:/MTd>
)

set_target_properties(MiniEldenTrainer PROPERTIES
    OUTPUT_NAME "MiniEldenTrainer"
    SUFFIX ".dll"
    PREFIX ""
)
